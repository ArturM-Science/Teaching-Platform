```python
    description=(
        "Review the research findings and verify the top 3 factual claims. "
        "Flag any claim that cannot be substantiated."
    ),

        "A numbered list of 3 claims. For each: the original claim, "
        "status (Confirmed / Disputed / Unverifiable), "
        "and a one-sentence evidence note explaining your verdict."

    agent=fact_checker,
    context=[research_task],
```

```python
from crewai.flow.flow import Flow, start, listen, router
from crews import research_crew, writing_crew

class ContentPipeline(Flow):

    @start()
    def research_phase(self):
        topic = self.state.get("topic", "AI")
        result = research_crew.kickoff(inputs={"topic": topic})
        return result.raw

    @listen(research_phase)
    def writing_phase(self, research_output):
        topic = self.state.get("topic", "AI")
        result = writing_crew.kickoff(inputs={"topic": topic})
        return result.raw

    @router(writing_phase)
    def quality_gate(self, draft):
        word_count = len(draft.split())
        print(f"\n[Quality Gate] Draft word count: {word_count}")
        return "publish" if word_count >= 150 else "revise"

    @listen("publish")
    def publish(self, content):
        word_count = len(content.split())
        print(f"\n[Flow] Published! Word count: {word_count}")
        return content

    @listen("revise")
    def revise(self, short_draft):
        word_count = len(short_draft.split())
        print(
            f"\n[Flow] Draft rejected â€” only {word_count} words "
            f"(minimum 150 required). Revision needed."
        )
        return "Revision needed."
```
