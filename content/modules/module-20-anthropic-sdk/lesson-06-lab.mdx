# Lab: Production Research Agent — Raw Anthropic SDK

You have learned the API contract, tool use, streaming, multi-agent patterns, and production reliability. Now you combine them. This lab is a capstone for the entire module: build a production-grade research agent using only the `anthropic` package — no LangChain, no LangGraph, no frameworks.

<LabBrief title="Production Research Agent — Raw Anthropic SDK" time="2–3 hours">
#### Objective
Build an async multi-tool research agent using ONLY the `anthropic` package. The agent must: use at least 3 tools (web search stub, calculator, file writer), support prompt caching on its system prompt, track cost per run, handle rate limits with exponential backoff, and support async parallel tool execution.

#### Setup
```bash
pip install anthropic tenacity pydantic
export ANTHROPIC_API_KEY="sk-ant-..."
```

#### Deliverables
- `agent.py` — async agent loop with tool dispatch and MAX_STEPS guard
- `tools.py` — 3 tool schemas + implementations + TOOL_MAP
- `cost_tracker.py` — accumulates usage across calls, prints summary at exit
- `prompts.py` — system prompt with `cache_control` block

#### Hard constraints
- Must use `anthropic.AsyncAnthropic()` — sync client fails the rubric
- Prompt caching must be enabled on the system prompt block
- Rate limit errors must be retried with exponential backoff using `tenacity`
- Cost tracker must print: input tokens, output tokens, cache hits, estimated USD
- All tool errors must return `{"error": "...", "retryable": bool}` — no unhandled exceptions
- `MAX_STEPS` must be ≤ 10
</LabBrief>

async def web_search(query: str) -> dict:
    """
    Real implementation using Brave Search API.
    Set BRAVE_API_KEY in your environment.
    pip install httpx
    """
    import os
    import httpx

    if not api_key:
        return {"error": "BRAVE_API_KEY not set", "retryable": False}

        response = await http_client.get(
            "https://api.search.brave.com/res/v1/web/search",
            params={"q": query, "count": 5},
            headers={"X-Subscription-Token": api_key, "Accept": "application/json"},
            timeout=10.0,
        )
        response.raise_for_status()
        data = response.json()
    
    results = []
    for item in data.get("web", {}).get("results", []):
        results.append({
            "title": item.get("title"),
            "snippet": item.get("description"),
            "url": item.get("url"),
        })
    
    return {"query": query, "results": results}

def record(self, usage) -> None:
    self.calls += 1
    self.input_tokens += usage.input_tokens
    self.output_tokens += usage.output_tokens
    self.cache_read_tokens += getattr(usage, "cache_read_input_tokens", 0)
    self.cache_write_tokens += getattr(usage, "cache_creation_input_tokens", 0)

def estimated_usd(self) -> float:
    p = self.PRICES.get(self.model, self.PRICES["claude-sonnet-4-6"])
    return (
        self.input_tokens * p["input"]
        + self.output_tokens * p["output"]
        + self.cache_read_tokens * p["cache_read"]
        + self.cache_write_tokens * p["cache_write"]
    )

# agent.py — complete solution
import asyncio
import json
import sys

import anthropic
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log,
)
import logging

from prompts import SYSTEM_PROMPT_BLOCKS
from tools import ALL_TOOL_SCHEMAS, ASYNC_TOOL_MAP
from cost_tracker import CostTracker

logging.basicConfig(level=logging.WARNING)
logger = logging.getLogger(__name__)

MODEL = "claude-sonnet-4-6"
MAX_STEPS = 10
tracker = CostTracker(model=MODEL)

# TODO 6 — solved
client = anthropic.AsyncAnthropic()


def make_resilient_create(async_client):
    # TODO 7 — solved
    @retry(
        wait=wait_exponential(multiplier=1, min=1, max=60),
        stop=stop_after_attempt(5),
        retry=retry_if_exception_type(anthropic.RateLimitError),
        before_sleep=before_sleep_log(logger, logging.WARNING),
    )
    async def create(**kwargs):
        return await async_client.messages.create(**kwargs)
    return create

resilient_create = make_resilient_create(client)


async def execute_tool(block) -> dict:
    if block.name not in ASYNC_TOOL_MAP:
        return {"error": f"Unknown tool: {block.name!r}", "retryable": False}
    try:
        return await ASYNC_TOOL_MAP[block.name](**block.input)
    except Exception as e:
        return {"error": str(e), "retryable": True}


async def run_agent(user_question: str) -> str:
    # TODO 8 — solved
    messages = [{"role": "user", "content": user_question}]

    for step in range(MAX_STEPS):
        response = await resilient_create(
            model=MODEL,
            max_tokens=4096,
            system=SYSTEM_PROMPT_BLOCKS,
            tools=ALL_TOOL_SCHEMAS,
            messages=messages,
        )
        tracker.record(response.usage)

        print(f"[Step {step + 1}] stop_reason={response.stop_reason}, "
              f"input={response.usage.input_tokens}, output={response.usage.output_tokens}")

        if response.stop_reason == "end_turn":
            text_blocks = [b for b in response.content if b.type == "text"]
            return text_blocks[0].text if text_blocks else ""

        if response.stop_reason != "tool_use":
            raise RuntimeError(f"Unexpected stop_reason: {response.stop_reason!r}")

        # Append assistant message (must come before tool_results)
        messages.append({"role": "assistant", "content": response.content})

        # Execute all tool calls in parallel
        tool_blocks = [b for b in response.content if b.type == "tool_use"]
        raw_results = await asyncio.gather(
            *[execute_tool(b) for b in tool_blocks],
            return_exceptions=True
        )

        tool_results = []
        for block, result in zip(tool_blocks, raw_results):
            if isinstance(result, Exception):
                result = {"error": str(result), "retryable": True}
            is_error = "error" in result
            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": json.dumps(result),
                **({"is_error": True} if is_error else {}),
            })
            print(f"  Tool {block.name!r}: {'ERROR' if is_error else 'OK'}")

        messages.append({"role": "user", "content": tool_results})

    raise RuntimeError(f"Agent did not converge after {MAX_STEPS} steps")


async def main():
    question = " ".join(sys.argv[1:]) if len(sys.argv) > 1 else (
        "Search for the latest AI benchmarks, calculate what percentage 90 is of 100, "
        "and write a summary report to 'ai_summary.txt'."
    )
    print(f"\nQuestion: {question}\n" + "-" * 60)
    try:
        answer = await run_agent(question)
        print("\n" + "=" * 60 + "\nAnswer:\n" + answer)
    except Exception as e:
        print(f"\nAgent failed: {e}")
        raise
    finally:
        tracker.print_summary()

if __name__ == "__main__":
    asyncio.run(main())
