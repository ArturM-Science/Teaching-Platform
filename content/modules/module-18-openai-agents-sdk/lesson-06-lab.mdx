# Lab: Customer Support System — Triage, Specialists, and Guardrails

You have learned the four primitives. Now you will build a production-shaped system that uses all of them together: a triage agent, two specialist agents with real tools, a regex PII guardrail, and an LLM-as-judge output guardrail.

<LabBrief title="Customer Support System: Triage, Specialists, and Guardrails" time="2–3 hours">
#### Objective
Build a three-agent customer support pipeline:

- **Triage agent** — reads user intent, routes to Billing or Tech Support using handoffs
- **Billing agent** — handles payment questions with invoice lookup and refund tools
- **Tech Support agent** — handles technical issues with diagnostic and status tools
- **Input guardrail** — reject messages containing PII (SSN, credit card, and one more pattern)
- **Output guardrail (LLM-as-judge)** — flag responses that make promises the company cannot keep

The system must handle tripwire exceptions gracefully and never expose internal error details to the user.


#### Deliverables
- `tools.py` — billing tools (`lookup_invoice`, `process_refund`) and tech tools (`run_diagnostic`, `check_status`)
- `guardrails.py` — PII input guardrail (3+ patterns) and LLM output guardrail
- `agents.py` — all three agents with tools and handoffs wired together
- `runner.py` — conversation loop that catches both tripwire exceptions and returns user-friendly messages
- `README.md` — one paragraph: when would you add a fourth agent instead of adding more tools to an existing agent?

#### Hard constraints
- Must use **handoffs** (not tools) for triage-to-specialist routing
- PII guardrail must check for at least 3 distinct patterns (SSN format, credit card format, and one more)
- Output guardrail must use a separate LLM agent — not regex
- `Runner.run()` must catch `InputGuardrailTripwireTriggered` and return a user-friendly message
- `RunConfig(max_turns=8)` must be set on every `Runner.run()` call
</LabBrief>

```python
# tools.py
from agents import function_tool, RunContextWrapper
from dataclasses import dataclass
from datetime import datetime, timezone

@dataclass
class SupportContext:
    user_id: str
    session_id: str

INVOICES = {
    "INV-001": {"amount": 99.00, "date": "2026-05-01", "status": "paid"},
    "INV-002": {"amount": 99.00, "date": "2026-06-01", "status": "duplicate"},
    "INV-003": {"amount": 299.00, "date": "2026-06-10", "status": "pending"},
}

DIAGNOSTICS = {
    "api": "API endpoints: all responding, avg latency 38ms",
    "database": "Database: connected, replication lag 0ms",
    "auth": "Auth service: degraded — elevated token refresh error rate (12%)",
    "storage": "Storage: all buckets accessible, no quota issues",
}

@function_tool
def lookup_invoice(ctx: RunContextWrapper[SupportContext], invoice_id: str) -> str:
    """Look up an invoice by ID and return its details including amount, date, and status."""
    inv = INVOICES.get(invoice_id.upper())
    if not inv:
        return f"Invoice {invoice_id} not found. Please check the invoice ID and try again."
    return (
        f"Invoice {invoice_id}: ${inv['amount']:.2f} charged on {inv['date']}, "
        f"status: {inv['status']}"
    )

@function_tool
def process_refund(ctx: RunContextWrapper[SupportContext], invoice_id: str, reason: str) -> str:
    """Process a refund for a given invoice. Returns confirmation with estimated timeline."""
    inv = INVOICES.get(invoice_id.upper())
    if not inv:
        return f"Cannot process refund: invoice {invoice_id} not found."
    print(f"AUDIT: refund initiated user={ctx.context.user_id} invoice={invoice_id}")
    return (
        f"Refund of ${inv['amount']:.2f} initiated for invoice {invoice_id}. "
        f"Reason: {reason}. Funds will appear in 3–5 business days."
    )

@function_tool
def run_diagnostic(ctx: RunContextWrapper[SupportContext], component: str) -> str:
    """Run a diagnostic check on a system component (api, database, auth, storage)."""
    result = DIAGNOSTICS.get(component.lower())
    if not result:
        valid = ", ".join(DIAGNOSTICS.keys())
        return f"Unknown component '{component}'. Valid options: {valid}"
    return result

@function_tool
def check_status(ctx: RunContextWrapper[SupportContext], service: str) -> str:
    """Check the operational status of a named service."""
    now = datetime.now(timezone.utc).strftime("%H:%M UTC")
    return f"Status for {service}: all systems operational as of {now}"
```

```python
# guardrails.py
import re
from agents import (
    Agent, Runner, InputGuardrail, OutputGuardrail,
    GuardrailFunctionOutput, RunContextWrapper
)
from pydantic import BaseModel

PII_PATTERNS = [
    r"\b\d{3}-\d{2}-\d{4}\b",                           # SSN: 123-45-6789
    r"\b\d{4}[- ]?\d{4}[- ]?\d{4}[- ]?\d{4}\b",         # Credit card: 16 digits
    r"\b(\+1\s?)?\(?\d{3}\)?[\s.-]\d{3}[\s.-]\d{4}\b",  # US phone number
]

async def pii_input_guardrail(
    ctx: RunContextWrapper,
    agent: Agent,
    input: str,
) -> GuardrailFunctionOutput:
    matched = [p for p in PII_PATTERNS if re.search(p, input)]
    return GuardrailFunctionOutput(
        output_info={"matched_patterns": matched, "input_length": len(input)},
        tripwire_triggered=bool(matched),
    )

class PromiseCheck(BaseModel):
    contains_promise: bool
    flagged_phrases: list[str]
    severity: str  # "none" | "low" | "high"

promise_checker_agent = Agent(
    name="PromiseChecker",
    model="gpt-4o-mini",
    instructions=(
        "You review customer support responses for promises the company cannot keep. "
        "Flag responses that contain: service guarantees ('we guarantee', '100% uptime'), "
        "data promises ('your data will never be lost'), unconditional refund promises "
        "('full refund no questions asked'), or SLA commitments with specific numbers. "
        "Mark severity as 'high' if the promise is specific and legally binding, "
        "'low' if it is vague, 'none' if there is no problematic promise."
    ),
    output_type=PromiseCheck,
)

async def promise_output_guardrail(
    ctx: RunContextWrapper,
    agent: Agent,
    output: str,
) -> GuardrailFunctionOutput:
    result = await Runner.run(promise_checker_agent, output, context=ctx.context)
    check = result.final_output
    return GuardrailFunctionOutput(
        output_info={
            "contains_promise": check.contains_promise,
            "flagged_phrases": check.flagged_phrases,
            "severity": check.severity,
        },
        tripwire_triggered=check.contains_promise and check.severity == "high",
    )

pii_guardrail = InputGuardrail(guardrail_function=pii_input_guardrail)
output_promise_guardrail = OutputGuardrail(guardrail_function=promise_output_guardrail)
```

```python
# runner.py
import asyncio
from agents import Runner, RunConfig
from agents.exceptions import InputGuardrailTripwireTriggered, OutputGuardrailTripwireTriggered
from agents import triage_agent  # assumes your file is agents.py; adjust import as needed
from tools import SupportContext

async def handle_message(user_id: str, session_id: str, message: str) -> str:
    ctx = SupportContext(user_id=user_id, session_id=session_id)
    try:
        result = await Runner.run(
            triage_agent,
            message,
            context=ctx,
            run_config=RunConfig(max_turns=8),
        )
        return result.final_output
    except InputGuardrailTripwireTriggered:
        return (
            "I'm not able to process messages that contain personal information "
            "such as social security numbers, credit card numbers, or phone numbers. "
            "Please remove any sensitive data and try again."
        )
    except OutputGuardrailTripwireTriggered:
        return (
            "I'm sorry, I was unable to provide a suitable response to your question. "
            "Please contact our support team directly at support@example.com for assistance."
        )

async def main():
    test_cases = [
        ("How do I cancel my subscription?", "general"),
```
